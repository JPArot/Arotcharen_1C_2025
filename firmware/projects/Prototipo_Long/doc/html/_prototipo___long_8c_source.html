164float calcularVelocidad() {
165 if (pulse_interval_us <= 0)
166 return 0;
167 float rps = 1000000.0f / (pulse_interval_us * SENSOR_SLOTS);
168 return 2 * M_PI * WHEEL_RADIUS * rps;
169}

176void frenar(void) {
177 for (int i = duty_cycle; i >= 0; i -= 10) {
178 L293SetSpeed(MOTOR_2, i);
179 vTaskDelay(pdMS_TO_TICKS(50));
180 }
181
182 duty_cycle = 0;
183
184 mantener_velocidad = false;
185 L293SetSpeed(MOTOR_2, 0);

191void tarea_comandos(void *param) {
192 while (1) {
193 if (cmd_frenar) {
194 frenar();
195 cmd_frenar = false;
196 }
197
198 if (mantener_velocidad) {
199 LedOn(LED_3); //Prendo LED rojo para indicar que se mantiene velocidad
200 mantener_velocidad=false;
201 if (duty_cycle < 255) {
202 duty_cycle += 10;
203 if (duty_cycle > 255) duty_cycle = 255;
204 L293SetSpeed(MOTOR_2, duty_cycle*100/255 );
205 }
206 }
207
208 vTaskDelay(pdMS_TO_TICKS(100));
209 }

215void tarea_monitoreo(void *param) {

227 float voltaje_bat = leerVoltajeBateria();

229 bateria_baja = true;
230 //frenar();
231 } else {
232 bateria_baja = false;
233 }
234
235 velocidad = calcularVelocidad();
236 snprintf(msg, sizeof(msg), "Bat: %.2f V | Vel: %.2f km/h", voltaje_bat, velocidad);
237 BleSendString(msg);
