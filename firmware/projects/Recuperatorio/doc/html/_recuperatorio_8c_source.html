101void TareaTeclas(void *param) {
102 bool estado_agua = true;
103 bool estado_comida = true;
104
105//controlo el agua
106 while (1) {
107 if (!GPIORead(TECLA_AGUA) && estado_agua) {
108 agua_habilitada = !agua_habilitada;
109 UartSendString(UART_CONSOLA, agua_habilitada ? "[AGUA ACTIVADA]\r\n" : "[AGUA DESACTIVADA]\r\n");
110 estado_agua = false; // cambio el estado de la bandera
111 }
112 if (GPIORead(TECLA_AGUA)) {
113 estado_agua = true;
114 }
115//uso la misma logica para la comida
116 if (!GPIORead(TECLA_COMIDA) && estado_comida) {
117 comida_habilitada = !comida_habilitada;
118 UartSendString(UART_CONSOLA, comida_habilitada ? "[COMIDA ACTIVADA]\r\n" : "[COMIDA DESACTIVADA]\r\n");
119 estado_comida = false;
120 }
121 if (GPIORead(TECLA_COMIDA)) {
122 estado_comida = true;
123 }
124
125 vTaskDelay(pdMS_TO_TICKS(100));
126 }
127}

135void TareaAgua(void *param) {
136 float distancia_cm;
137 while (1) {
138 if (agua_habilitada) {
139 if (HcSr04ReadDistance(&distancia_cm)) {
140 agua_ml = (30.0 - distancia_cm) * 100;
141 if (agua_ml < 500) {
142 GPIOOn(ELECTROVALVULA);
143 } else if (agua_ml >= 2500) {
144 GPIOOff(ELECTROVALVULA);
145 }
146 }
147 } else {
148 GPIOOff(ELECTROVALVULA);
149 }
150 vTaskDelay(pdMS_TO_TICKS(2000));
151 }
152}

161void TareaComida(void *param) {
162 uint16_t raw;
163 while (1) {
164 if (comida_habilitada) {
165 AnalogInputReadSingle(CH2, &raw);
166 float volt = (raw / 4095.0f) * 3.3f;
167 comida_g = ((volt - 0.5f) / 2.8f) * 500.0f; // (3.3-0.5)=2.8 (que seria la variacion de voltaje maximo y minimo)
168 //si la comida es menor a 25g, activo el motor de comida
169 if (comida_g < 25) {
170 GPIOOn(MOTOR_COMIDA);
171 //si la comida es mayor a 250g, desactivo el motor de comida
172 } else if (comida_g >= 250) {
173 GPIOOff(MOTOR_COMIDA);
174 }
175 } else {
176 GPIOOff(MOTOR_COMIDA);
177 }
178 vTaskDelay(pdMS_TO_TICKS(5000));
179 }
180}

187void TareaUART(void *param) {
188 char comando;
189 while (1) {
190 if (UartReadByte(UART_CONSOLA, (uint8_t *)&comando)) {
191 if (comando == 'w') {
192 agua_habilitada = !agua_habilitada;
193 UartSendString(UART_CONSOLA, agua_habilitada ? "[AGUA ACTIVADA]\r\n" : "[AGUA DESACTIVADA]\r\n");
194 }
195 if (comando == 'f') {
196 comida_habilitada = !comida_habilitada;
197 UartSendString(UART_CONSOLA, comida_habilitada ? "[COMIDA ACTIVADA]\r\n" : "[COMIDA DESACTIVADA]\r\n");
198 }
199 }
200 vTaskDelay(pdMS_TO_TICKS(100));
201 }
202}
